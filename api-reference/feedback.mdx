## recordFeedback

```ts
recordFeedback(options): Promise<{ accepted: number }>
```

<ParamField path="apiKey" type="string" required>
  Project API key. Keep it server-side.
</ParamField>

<ParamField path="target" type="object" required>
  What the feedback attaches to.
  <Expandable title="target">
    <ParamField path="type" type='"message" | "session"' required>
      A turn (`message`) or a conversation (`session`).
    </ParamField>
    <ParamField path="id" type="string" required>
      The `messageId` (message) or `sessionId` (session).
    </ParamField>
    <ParamField path="spanId" type="string">
      A specific sub-step span of a turn.
    </ParamField>
  </Expandable>
</ParamField>

<ParamField path="kind" type='"thumb" | "score" | "comment" | "correction" | "label"' required>
  The shape of the signal.
</ParamField>

<ParamField path="score" type="number">
  `1` / `0` for a thumb, or a scalar.
</ParamField>

<ParamField path="label" type="string">
  Categorical tag, or a raw thumb token.
</ParamField>

<ParamField path="comment" type="string">
  Free text; also where a `correction` payload goes.
</ParamField>

<ParamField path="name" type="string" default="user">
  The feedback dimension (e.g. `helpfulness`). Record multiple named signals per target.
</ParamField>

<ParamField path="source" type='"end_user" | "labeler" | "model_judge" | "code"' default="end_user">
  Who or what produced the signal.
</ParamField>

<ParamField path="userId" type="string">
  The subject of the run; also the default annotator.
</ParamField>

<ParamField path="annotatorId" type="string" default="userId">
  Who gave the feedback (a reviewer or labeler). Defaults to `userId`.
</ParamField>

<ParamField path="sessionId" type="string">
  Denormalized session id, for `message` targets.
</ParamField>

<ParamField path="metadata" type="Record<string, string>">
  Arbitrary string tags stored with the signal.
</ParamField>

<ParamField path="id" type="string">
  Idempotency id. Server-generated if omitted; latest write wins.
</ParamField>

<ParamField path="endpoint" type="string" default="https://ingest.eu.bryel.ai/v1/feedback">
  Override the feedback endpoint when self-hosting.
</ParamField>

<Note>
  Writes are latest-wins per `(target, name, source, annotator, kind)` — re-recording the same dimension updates it rather than duplicating.
</Note>

## Shorthands

```ts
thumbsUp(options)    // kind: "thumb", score: 1
thumbsDown(options)  // kind: "thumb", score: 0
recordFeedbackBatch({ apiKey }, items) // record many in one POST
```
